```python
from __future__ import annotations
import importlib.util
import os
import subprocess
import sys
from pathlib import Path

def run_lint(wiki_dir: Path, *args: str) -> subprocess.CompletedProcess:
    env = {**os.environ, "DND_DIR": str(wiki_dir)}
    return subprocess.run(
        [sys.executable, str(LINT), *args],
        capture_output=True, text=True, env=env,

def test_clean_wiki_has_no_errors(wiki):
    r = run_lint(wiki, "--format", "json")
    import json
    data = json.loads(r.stdout)
    assert data["summary"]["errors"] == 0, r.stdout


def test_missing_required_field_detected(wiki):
    target = wiki / "npcs" / "npc-one.md"
    text = target.read_text()
    # Drop `race:` to trigger missing-required
    text = "\n".join(l for l in text.splitlines() if not l.startswith("race:"))
    target.write_text(text)
    r = run_lint(wiki, "--only", "missing-required", "--format", "json")

    data = json.loads(r.stdout)
```

```python
def _levenshtein(a: str, b: str) -> int:
    if len(a) < len(b):
        return _levenshtein(b, a)
    if not b:
        return len(a)
    prev = list(range(len(b) + 1))
    for i, ca in enumerate(a):
        cur = [i + 1]
        for j, cb in enumerate(b):
            cur.append(min(cur[j] + 1, prev[j + 1] + 1, prev[j] + (ca != cb)))
        prev = cur
    return prev[-1]


@register("taxonomy-hygiene")
def check_taxonomy_hygiene(*, dnd_dir, schema, pages) -> list[Issue]:
    """Run against SCHEMA.md itself: near-dups, prefix-containment, cross-cat."""
    schema_path = dnd_dir / "SCHEMA.md"
    taxonomy = schema.get("taxonomy", {}) or {}
    if not isinstance(taxonomy, dict):
        return []
    out = []

    # Near-dup and prefix-containment within each category
    for cat, tags in taxonomy.items():
        if not isinstance(tags, list):
            continue
        tag_list = [t for t in tags if isinstance(t, str)]
        for i in range(len(tag_list)):
            for j in range(i + 1, len(tag_list)):
                a, b = tag_list[i], tag_list[j]
                if a == b:
                    continue
                # prefix-containment: one is a hyphen-extension of the other
                if (a + "-" == b[: len(a) + 1]) or (b + "-" == a[: len(b) + 1]):
                    out.append(Issue(
                        check="taxonomy-hygiene", severity="info",
                        file=schema_path, field=cat,
                        message=f"prefix-containment in '{cat}': '{a}' ⊂ '{b}' (or vice versa)",
                        fixable=False,
                    ))
                    continue
                dist = _levenshtein(a, b)
                if dist <= 2 and abs(len(a) - len(b)) <= 2 and min(len(a), len(b)) >= 4:
                    out.append(Issue(
                        check="taxonomy-hygiene", severity="info",
                        file=schema_path, field=cat,
                        message=f"near-dup in '{cat}': '{a}' ↔ '{b}' (levenshtein={dist})",
                        fixable=False,
                    ))

    # Cross-category appearances
    tag_cats: dict[str, list[str]] = {}
    for cat, tags in taxonomy.items():
        if not isinstance(tags, list):
            continue
        for t in tags:
            if isinstance(t, str):
                tag_cats.setdefault(t, []).append(cat)
    for t, cats in tag_cats.items():
        if len(cats) > 1:
            out.append(Issue(
                check="taxonomy-hygiene", severity="info",
                file=schema_path, field="taxonomy",
                message=f"cross-category: '{t}' appears in {cats}",
                fixable=False,
            ))
    return out
```

```python
        race = c.get("race", "?")
        classes = c.get("classes") or []
        cls = classes[0] if isinstance(classes, list) and classes else "?"
        lines.append(f"| {player} | {name} | {slug} | {race} | {cls} |")
    return "\n".join(lines)


def replace_block(text: str, begin: str, end: str, new_body: str) -> tuple[str, bool]:
    """Return (new_text, changed). Raises ValueError if markers missing."""
    pattern = re.compile(
        re.escape(begin) + r"\n(.*?)\n" + re.escape(end),
        re.S,
    )
    m = pattern.search(text)
    if not m:
        raise ValueError(f"markers missing: {begin!r} / {end!r}")
    current = m.group(1)
    if current == new_body:
        return text, False
    new_block = f"{begin}\n{new_body}\n{end}"
    return text[:m.start()] + new_block + text[m.end():], True
```

```python
def migrate_state_file(dnd_dir: Path) -> bool:
    """Insert party markers into STATE.md after '## Партия сейчас' heading.

    Idempotent: if markers already present, no-op.
    """
    path = dnd_dir / "STATE.md"
    if not path.exists():
        return False
    text = path.read_text(encoding="utf-8")
    if STATE_PARTY_BEGIN in text:
        return False
    # Insert empty block after '## Партия' heading or at end of body
    marker_block = f"\n{STATE_PARTY_BEGIN}\n\n{STATE_PARTY_END}\n"
    if "## Партия" in text:
        idx = text.find("## Партия")
        # find end of this section (next '##' or EOF)
        next_h = text.find("\n## ", idx + 4)
        insert_at = next_h if next_h != -1 else len(text)
        text = text[:insert_at] + marker_block + text[insert_at:]
    else:
        text = text.rstrip() + "\n\n## Партия сейчас\n" + marker_block
    path.write_text(text, encoding="utf-8")
    return True
```

```python
def _load_sync_state_module():
    """Import dnd-sync-state.py as a module for in-process drift checks.

    Returns the module or None if the script doesn't exist.
    """
    import importlib.util
    path = SCRIPT_DIR / "dnd-sync-state.py"
    if not path.exists():
        return None
    spec = importlib.util.spec_from_file_location("dnd_sync_state", path)
    if spec is None or spec.loader is None:
        return None
    mod = importlib.util.module_from_spec(spec)
    spec.loader.exec_module(mod)
    return mod


@register("state-drift")
def check_state_drift(*, dnd_dir, schema, pages) -> list[Issue]:
    path = dnd_dir / "STATE.md"
    if not path.exists():
        return []
    mod = _load_sync_state_module()
    if mod is None:
        return []
    chars = mod.collect_characters(dnd_dir)
    quest_status = mod.collect_quest_statuses(dnd_dir)
    try:
        body_drift = mod.sync_state_file(dnd_dir, chars, check_only=True)
        fm_drift = mod.sync_state_frontmatter(dnd_dir, quest_status, check_only=True)
    except SystemExit:
        return [Issue(
            check="state-drift", severity="error",
            file=path, field=None,
            message="STATE.md missing markers; run `dnd-sync-state.py --migrate`",
            fixable=False,
        )]
    if body_drift or fm_drift:
        return [Issue(
            check="state-drift", severity="error",
            file=path, field=None,
            message="derived blocks differ from source of truth; run `dnd-sync-state.py --fix`",
            fixable=True,
        )]
    return []


@register("campaign-drift")
def check_campaign_drift(*, dnd_dir, schema, pages) -> list[Issue]:
    path = dnd_dir / "CAMPAIGN.md"
    if not path.exists():
        return []
    mod = _load_sync_state_module()
    if mod is None:
        return []
    chars = mod.collect_characters(dnd_dir)
    try:
        drifted = mod.sync_campaign_file(dnd_dir, chars, check_only=True)
    except SystemExit:
        return [Issue(
            check="campaign-drift", severity="error",
            file=path, field=None,
            message="CAMPAIGN.md missing markers; run `dnd-sync-state.py --migrate`",
            fixable=False,
        )]
    if drifted:
        return [Issue(
            check="campaign-drift", severity="error",
            file=path, field=None,
            message="players table drift; run `dnd-sync-state.py --fix`",
            fixable=True,
        )]
    return []
```
